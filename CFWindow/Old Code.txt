public enum Themes
    {
        Windows8Blue, Windows8Grey, Windows8LightBlue, Windows8LightPink, Windows8Yellow,
        Windows8Lime, Windows8Turquoise, Windows8Orange, Windows8Red, Windows8Pink,
        Windows8Green, Windows8Purple, Windows8Lilac, Windows8Brown, Windows8White, Windows8Inactive
    };

	public SolidColorBrush GetResourceColour(Themes theme, string key, string style)
        {
            return GetResource<SolidColorBrush>(theme.ToString().Replace(style, "") + "." + key);
        }

		public T GetResource<T>(string key)
        {
            return (T)Application.Current.Resources[key];
        }
